Option Explicit

' ================== CONFIG ==================
Private Const TARGET_MAILBOX As String = "facturas@novogar.cr"      ' Puede ser DisplayName o SMTP (depende del Outlook)
Private Const INBOX_NAME As String = "Bandeja de entrada"
Private Const TARGET_FOLDER_NAME As String = "FACTURAS RECIBIDAS"  ' Subcarpeta dentro de Bandeja de entrada (o se busca en todo el buzón)
Private Const SAVE_FOLDER As String = "C:\Jose\Proyecto Novogar\documentos\facturas recibidas\"
' ============================================

' ================== ENTRYPOINT (recomendado) ==================
' 1) Te pide rango (dd/mm/yyyy hh:mm 24h)
' 2) Te deja elegir el folder real:
'    - Si tienes un correo seleccionado, usa EXACTAMENTE la carpeta real de ese correo (lo más robusto).
'    - Si no hay selección, intenta por ruta (Inbox\FACTURAS RECIBIDAS) y si no, busca por nombre en todo el buzón.
Public Sub DescargarAdjuntos_FACTURAS_RECIBIDAS_RangoFecha()
    Dim startDate As Date, endDate As Date
    Dim targetFolder As Outlook.Folder

    On Error GoTo EH

    If Dir(SAVE_FOLDER, vbDirectory) = "" Then
        MsgBox "La carpeta de destino no existe: " & SAVE_FOLDER, vbCritical
        Exit Sub
    End If

    If Not PromptDateRange(startDate, endDate) Then Exit Sub

    Set targetFolder = ResolveTargetFolderRobust()
    If targetFolder Is Nothing Then
        MsgBox "No pude resolver la carpeta destino." & vbCrLf & _
               "Tip: selecciona un correo dentro de la carpeta REAL y vuelve a ejecutar.", vbCritical
        Exit Sub
    End If

    ProcessFolderByDateRange targetFolder, startDate, endDate
    Exit Sub

EH:
    MsgBox "Error: " & Err.Number & " - " & Err.Description, vbCritical
End Sub

' ================== DEBUG: imprime la carpeta real del correo seleccionado ==================
Public Sub DEBUG_CarpetaRealDelCorreoSeleccionado()
    Dim sel As Outlook.Selection
    Dim it As Object
    Dim mail As Outlook.MailItem

    If sel Is Nothing Or sel.Count = 0 Then

    Debug.Print "EntryID:", mail.entryId
    Debug.Print "============================="

    MsgBox "Listo. Revisa Ctrl+G (Immediate Window).", vbInformation
End Sub

' ================== CORE PROCESSING ==================
Private Sub ProcessFolderByDateRange(ByVal targetFolder As Outlook.Folder, ByVal startDate As Date, ByVal endDate As Date)
    Dim items As Outlook.items
    Dim it As Object
    Dim mail As Outlook.MailItem

    Dim countMails As Long, countFiles As Long, countSkippedRule As Long
    Dim scanned As Long

    On Error GoTo EH

    Set items = targetFolder.items
    items.Sort "[ReceivedTime]", True ' más nuevos primero

    Debug.Print "==== PROCESO ===="
    Debug.Print "Folder:", targetFolder.folderPath
    Debug.Print "Items total:", items.Count
    Debug.Print "Start:", startDate, " End:", endDate
    Debug.Print "==============="

    countMails = 0
    countFiles = 0
    countSkippedRule = 0

    ' Nota: NO usamos Restrict para evitar problemas de locale/vistas.
    ' Recorremos en orden descendente y salimos temprano cuando pasamos el rango.
    For Each it In items
        scanned = scanned + 1

        If TypeOf it Is Outlook.MailItem Then
            Set mail = it

            If mail.ReceivedTime > endDate Then
                ' Aún más nuevo que el rango -> seguir bajando
            ElseIf mail.ReceivedTime < startDate Then
                ' Ya pasamos el rango (como está ordenado desc, podemos salir)
                Exit For
            Else
                ' Dentro del rango
                If mail.Attachments.Count > 0 Then
                    If HasMinimumPdfXmlAttachments(mail) Then
                        countMails = countMails + 1
                        countFiles = countFiles + SaveAttachments_PdfXml_WithManifest(mail)
                    Else
                        countSkippedRule = countSkippedRule + 1
                    End If
                End If
            End If
        End If

        ' Evitar loops eternos si hay carpetas gigantes y rango muy reciente:
        ' (opcional) puedes comentar esto si no lo quieres
        If scanned >= 200000 Then Exit For
    Next it

    MsgBox "Proceso finalizado." & vbCrLf & _
           "Carpeta: " & targetFolder.folderPath & vbCrLf & _
           "Correos procesados (>=1 PDF y >=2 XML): " & countMails & vbCrLf & _
           "Correos omitidos por regla mínima: " & countSkippedRule & vbCrLf & _
           "Archivos guardados (solo PDF/XML): " & countFiles & vbCrLf & _
           "Destino: " & SAVE_FOLDER, vbInformation
    Exit Sub

EH:
    MsgBox "Error procesando carpeta: " & Err.Number & " - " & Err.Description, vbCritical
End Sub

' ================== Folder Resolution (robusto) ==================
' Prioridad:
' A) Si hay un correo seleccionado -> usar su carpeta real (lo más confiable)
' B) Intentar ruta: TARGET_MAILBOX\Bandeja de entrada\FACTURAS RECIBIDAS
' C) Buscar por nombre en todo el buzón
Private Function ResolveTargetFolderRobust() As Outlook.Folder
    Dim fromSelection As Outlook.Folder
    Dim byPath As Outlook.Folder
    Dim bySearch As Outlook.Folder

    Set fromSelection = GetSelectedMailParentFolder()
    If Not fromSelection Is Nothing Then
        Debug.Print "Usando carpeta desde selección:", fromSelection.folderPath
        Set ResolveTargetFolderRobust = fromSelection
        Exit Function
    End If

    Set byPath = GetFolderByInboxPath()
    If Not byPath Is Nothing Then
        Debug.Print "Usando carpeta por ruta Inbox:", byPath.folderPath
        Set ResolveTargetFolderRobust = byPath
        Exit Function
    End If

    Set bySearch = FindFolderRecursive(GetStoreRoot(), TARGET_FOLDER_NAME)
    If Not bySearch Is Nothing Then
        Debug.Print "Usando carpeta por búsqueda global:", bySearch.folderPath
        Set ResolveTargetFolderRobust = bySearch
        Exit Function
    End If

    Set ResolveTargetFolderRobust = Nothing
End Function

Private Function GetSelectedMailParentFolder() As Outlook.Folder
    On Error GoTo EH
    Dim sel As Outlook.Selection
    Dim it As Object
    Dim mail As Outlook.MailItem

    Set sel = Application.ActiveExplorer.Selection
    If sel Is Nothing Or sel.Count = 0 Then
        Set GetSelectedMailParentFolder = Nothing
        Exit Function
    End If

    Set it = sel.item(1)
    If Not TypeOf it Is Outlook.MailItem Then
        Set GetSelectedMailParentFolder = Nothing
        Exit Function
    End If

    Set mail = it
    Set GetSelectedMailParentFolder = mail.parent
    Exit Function

EH:
    Set GetSelectedMailParentFolder = Nothing
End Function

Private Function GetFolderByInboxPath() As Outlook.Folder
    On Error GoTo EH
    Dim ns As Outlook.NameSpace
    Dim storeRoot As Outlook.Folder
    Dim inbox As Outlook.Folder
    Dim targetFolder As Outlook.Folder

    Set ns = Application.GetNamespace("MAPI")
    Set storeRoot = ns.Folders(TARGET_MAILBOX)
    Set inbox = storeRoot.Folders(INBOX_NAME)
    Set targetFolder = GetSubFolderByName(inbox, TARGET_FOLDER_NAME)

    Set GetFolderByInboxPath = targetFolder
    Exit Function

EH:
    Set GetFolderByInboxPath = Nothing
End Function

Private Function GetStoreRoot() As Outlook.Folder
    Dim ns As Outlook.NameSpace
    Set ns = Application.GetNamespace("MAPI")
    Set GetStoreRoot = ns.Folders(TARGET_MAILBOX)
End Function

Private Function FindFolderRecursive(ByVal parent As Outlook.Folder, ByVal folderName As String) As Outlook.Folder
    Dim f As Outlook.Folder

    If StrComp(parent.name, folderName, vbTextCompare) = 0 Then
        Set FindFolderRecursive = parent
        Exit Function
    End If

    For Each f In parent.Folders
        Set FindFolderRecursive = FindFolderRecursive(f, folderName)
        If Not FindFolderRecursive Is Nothing Then Exit Function
    Next f

    Set FindFolderRecursive = Nothing
End Function

' ================== Date Prompt ==================
Private Function PromptDateRange(ByRef startDate As Date, ByRef endDate As Date) As Boolean
    Dim inputStart As String, inputEnd As String

    inputStart = InputBox( _
        "Ingrese fecha y hora INICIAL en formato:" & vbCrLf & _
        "dd/mm/yyyy hh:mm (24 horas)" & vbCrLf & _
        "Ejemplo: 27/02/2026 09:30", _
        "Rango Inicial" _
    )
    If inputStart = "" Then
        PromptDateRange = False
        Exit Function
    End If

    inputEnd = InputBox( _
        "Ingrese fecha y hora FINAL en formato:" & vbCrLf & _
        "dd/mm/yyyy hh:mm (24 horas)" & vbCrLf & _
        "Ejemplo: 02/03/2026 14:16", _
        "Rango Final" _
    )
    If inputEnd = "" Then
        PromptDateRange = False
        Exit Function
    End If

    If Not IsDate(inputStart) Or Not IsDate(inputEnd) Then
        MsgBox "Formato de fecha inválido. Use: dd/mm/yyyy hh:mm (24h)." & vbCrLf & _
               "Ejemplo: 27/02/2026 09:30", vbCritical
        PromptDateRange = False
        Exit Function
    End If

    startDate = CDate(inputStart)
    endDate = CDate(inputEnd)

    If endDate < startDate Then
        MsgBox "Rango inválido: la fecha/hora final es menor que la inicial.", vbCritical
        PromptDateRange = False
        Exit Function
    End If

    PromptDateRange = True
End Function

' ================== CORE: Save attachments + Manifest ==================
' Guarda SOLO PDF y XML (ignora otros), clasifica XML (DOC vs RH) y crea manifest.json
' Nombres:
' <ingestion_id>.PDF.1.pdf
' <ingestion_id>.DOC.1.xml
' <ingestion_id>.RH.1.xml
' <ingestion_id>.manifest.json
Private Function SaveAttachments_PdfXml_WithManifest(ByVal mail As Outlook.MailItem) As Long
    Dim att As Outlook.Attachment
    Dim ingestionId As String
    Dim ext As String
    Dim saved As Long

    Dim pdfIdx As Long, docIdx As Long, rhIdx As Long
    Dim tmpPath As String, finalName As String, finalPath As String, kind As String

    Dim manifestItems As String
    Dim firstItem As Boolean

    ingestionId = BuildIngestionId(mail)
    firstItem = True
    manifestItems = ""

    For Each att In mail.Attachments
        ext = LCase$(GetExtension(att.fileName))

        If ext <> "pdf" And ext <> "xml" Then GoTo SkipAtt

        If ext = "pdf" Then
            pdfIdx = pdfIdx + 1
            finalName = ingestionId & ".PDF." & pdfIdx & ".pdf"
            finalPath = EnsureUniquePath(SAVE_FOLDER & finalName)
            att.SaveAsFile finalPath

            AppendManifestItem manifestItems, firstItem, "PDF", finalName, att.fileName, att.size
            saved = saved + 1

        ElseIf ext = "xml" Then
            tmpPath = EnsureUniquePath(SAVE_FOLDER & ingestionId & ".__TMP__." & att.Index & ".xml")
            att.SaveAsFile tmpPath

            kind = ClassifyXmlByRoot(tmpPath) ' "DOC", "RH", "UNKNOWN"

            If kind = "DOC" Then
                docIdx = docIdx + 1
                finalName = ingestionId & ".DOC." & docIdx & ".xml"
            ElseIf kind = "RH" Then
                rhIdx = rhIdx + 1
                finalName = ingestionId & ".RH." & rhIdx & ".xml"
            Else
                On Error Resume Next
                Kill tmpPath
                On Error GoTo 0
                GoTo SkipAtt
            End If

            finalPath = EnsureUniquePath(SAVE_FOLDER & finalName)
            Name tmpPath As finalPath

            AppendManifestItem manifestItems, firstItem, kind, finalName, att.fileName, att.size
            saved = saved + 1
        End If

SkipAtt:
    Next att

    If saved > 0 Then
        WriteManifestJson SAVE_FOLDER & ingestionId & ".manifest.json", ingestionId, mail, manifestItems
    End If

    SaveAttachments_PdfXml_WithManifest = saved
End Function

' ================== XML classification ==================
Private Function ClassifyXmlByRoot(ByVal xmlPath As String) As String
    Dim xml As Object ' MSXML2.DOMDocument.6.0
    Dim root As Object
    Dim rootName As String, ns As String

    On Error GoTo EH

    Set xml = CreateObject("MSXML2.DOMDocument.6.0")
    xml.Async = False
    xml.ValidateOnParse = False
    xml.ResolveExternals = False

    If Not xml.Load(xmlPath) Then
        ClassifyXmlByRoot = "UNKNOWN"
        Exit Function
    End If

    Set root = xml.DocumentElement
    If root Is Nothing Then
        ClassifyXmlByRoot = "UNKNOWN"
        Exit Function
    End If

    rootName = LCase$(root.BaseName)
    ns = LCase$(root.NamespaceURI)

    If rootName = "mensajehacienda" And InStr(1, ns, "mensajehacienda", vbTextCompare) > 0 Then
        ClassifyXmlByRoot = "RH"
        Exit Function
    End If

    If IsElectronicDocRoot(rootName) Then
        ClassifyXmlByRoot = "DOC"
        Exit Function
    End If

    ClassifyXmlByRoot = "UNKNOWN"
    Exit Function

EH:
    ClassifyXmlByRoot = "UNKNOWN"
End Function

Private Function IsElectronicDocRoot(ByVal rootNameLower As String) As Boolean
    Select Case rootNameLower
        Case "facturaelectronica", _
             "tiqueteelectronico", _
             "notacreditoelectronica", _
             "notadebitoelectronica", _
             "facturacompra"
            IsElectronicDocRoot = True
        Case Else
            IsElectronicDocRoot = False
    End Select
End Function

' ================== Manifest JSON ==================
Private Sub AppendManifestItem(ByRef jsonArr As String, ByRef firstItem As Boolean, _
                               ByVal tipo As String, ByVal savedAs As String, _
                               ByVal originalName As String, ByVal sizeBytes As Long)
    Dim item As String
    item = "{""type"":""" & JsonEscape(tipo) & """," & _
           """savedAs"":""" & JsonEscape(savedAs) & """," & _
           """originalName"":""" & JsonEscape(originalName) & """," & _
           """sizeBytes"":" & CStr(sizeBytes) & "}"

    If firstItem Then
        jsonArr = item
        firstItem = False
    Else
        jsonArr = jsonArr & "," & item
    End If
End Sub

Private Sub WriteManifestJson(ByVal manifestPath As String, ByVal ingestionId As String, _
                              ByVal mail As Outlook.MailItem, ByVal attachmentsJson As String)
    Dim f As Integer
    Dim sender As String, subject As String, entryId As String, receivedIso As String

    sender = SafeSenderEmail(mail)
    subject = mail.subject
    entryId = mail.entryId
    receivedIso = Format$(mail.ReceivedTime, "yyyy-mm-dd\THH:nn:ss")

    f = FreeFile
    Open manifestPath For Output As #f

    Print #f, "{"
    Print #f, "  ""ingestion_id"": """ & JsonEscape(ingestionId) & ""","
    Print #f, "  ""source"": ""outlook"","
    Print #f, "  ""received_time"": """ & JsonEscape(receivedIso) & ""","
    Print #f, "  ""sender"": """ & JsonEscape(sender) & ""","
    Print #f, "  ""subject"": """ & JsonEscape(subject) & ""","
    Print #f, "  ""outlook_entry_id"": """ & JsonEscape(entryId) & ""","
    Print #f, "  ""attachments_saved"": [" & attachmentsJson & "]"
    Print #f, "}"

    Close #f
End Sub

Private Function SafeSenderEmail(ByVal mail As Outlook.MailItem) As String
    On Error GoTo EH
    SafeSenderEmail = mail.SenderEmailAddress
    Exit Function
EH:
    SafeSenderEmail = ""
End Function

Private Function JsonEscape(ByVal s As String) As String
    s = Replace(s, "\", "\\")
    s = Replace(s, """", "\""")
    s = Replace(s, vbCrLf, "\n")
    s = Replace(s, vbCr, "\n")
    s = Replace(s, vbLf, "\n")
    JsonEscape = s
End Function

' ================== Utilities ==================
Private Function GetSubFolderByName(ByVal parentFolder As Outlook.Folder, ByVal subFolderName As String) As Outlook.Folder
    Dim f As Outlook.Folder
    For Each f In parentFolder.Folders
        If StrComp(f.name, subFolderName, vbTextCompare) = 0 Then
            Set GetSubFolderByName = f
            Exit Function
        End If
    Next f
    Set GetSubFolderByName = Nothing
End Function

Private Function GetExtension(ByVal fileName As String) As String
    Dim p As Long
    p = InStrRev(fileName, ".")
    If p <= 0 Then
        GetExtension = ""
    Else
        GetExtension = Mid$(fileName, p + 1)
    End If
End Function

' Regla mínima: al menos 1 PDF y 2 XML
Private Function HasMinimumPdfXmlAttachments(ByVal mail As Outlook.MailItem) As Boolean
    Dim att As Outlook.Attachment
    Dim ext As String
    Dim pdfCount As Long, xmlCount As Long

    For Each att In mail.Attachments
        ext = LCase$(GetExtension(att.fileName))
        If ext = "pdf" Then
            pdfCount = pdfCount + 1
        ElseIf ext = "xml" Then
            xmlCount = xmlCount + 1
        End If
    Next att

    HasMinimumPdfXmlAttachments = (pdfCount >= 1 And xmlCount >= 2 And (pdfCount + xmlCount) >= 3)
End Function

' Ingestion ID estable: timestamp + cola del EntryID alfanumérico
Private Function BuildIngestionId(ByVal mail As Outlook.MailItem) As String
    Dim ts As String, tail As String
    ts = Format$(mail.ReceivedTime, "yyyymmdd_hhnnss")
    tail = Right$(CleanId(mail.entryId), 18)
    If tail = "" Then tail = "NOENTRYID"
    BuildIngestionId = ts & "_" & tail
End Function

Private Function CleanId(ByVal s As String) As String
    Dim i As Long, ch As String, out As String
    out = ""
    For i = 1 To Len(s)
        ch = Mid$(s, i, 1)
        If (ch >= "0" And ch <= "9") Or (ch >= "A" And ch <= "Z") Or (ch >= "a" And ch <= "z") Then
            out = out & ch
        End If
    Next i
    CleanId = out
End Function

' Evita colisiones: agrega (1), (2), ...
Private Function EnsureUniquePath(ByVal fullPath As String) As String
    Dim base As String, ext As String, i As Long, dotPos As Long, candidate As String

    If Dir(fullPath) = "" Then
        EnsureUniquePath = fullPath
        Exit Function
    End If

    dotPos = InStrRev(fullPath, ".")
    If dotPos > 0 Then
        base = Left$(fullPath, dotPos - 1)
        ext = Mid$(fullPath, dotPos)
    Else
        base = fullPath
        ext = ""
    End If

    i = 1
    Do
        candidate = base & " (" & i & ")" & ext
        If Dir(candidate) = "" Then
            EnsureUniquePath = candidate
            Exit Function
        End If
        i = i + 1
    Loop
End Function
